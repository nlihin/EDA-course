def classify_group(row):
    if row['Age'] >= 18 and row['Age'] <= 21:
        return 'Female 18-21' if row['Gender'] == 'Female' else 'Male 18-21'
    elif row['Age'] > 21 and row['Age'] <= 25:
        return 'Female 21-25' if row['Gender'] == 'Female' else 'Male 21-25'
    else:
        return None  #  

#  
df['Age_Gender_Group'] = df.apply(classify_group, axis=1)
